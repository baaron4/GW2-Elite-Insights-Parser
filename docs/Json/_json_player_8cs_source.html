221  private static List<JsonPlayerBuffsGeneration> GetPlayerBuffGenerations(List<Dictionary<long, FinalPlayerBuffs>> buffs, ParsedLog log, Dictionary<string, JsonLog.BuffDesc> buffDesc)
222  {
223  List<PhaseData> phases = log.FightData.GetPhases(log);
224  var uptimes = new List<JsonPlayerBuffsGeneration>();
225  foreach (KeyValuePair<long, FinalPlayerBuffs> pair in buffs[0])
226  {
227  Buff buff = log.Buffs.BuffsByIds[pair.Key];
228  if (!buffDesc.ContainsKey("b" + pair.Key))
229  {
230  buffDesc["b" + pair.Key] = new JsonLog.BuffDesc(buff);
231  }
232  var data = new List<JsonBuffsGenerationData>();
233  for (int i = 0; i < phases.Count; i++)
234  {
235  data.Add(new JsonBuffsGenerationData(buffs[i][pair.Key]));
236  }
237  var jsonBuffs = new JsonPlayerBuffsGeneration()
238  {
239  BuffData = data,
240  Id = pair.Key
241  };
242  uptimes.Add(jsonBuffs);
243  }
244 
245  if (!uptimes.Any())
246  {
247  return null;
248  }
249 
250  return uptimes;
251  }
